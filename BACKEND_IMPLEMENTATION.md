Security Module\n\n```python\n# app/core/security.py\nfrom datetime import datetime, timedelta\nfrom typing import Optional\nfrom jose import JWTError, jwt\nfrom passlib.context import CryptContext\nfrom app.config import settings\n\npwd_context = CryptContext(schemes=[\"bcrypt\"], deprecated=\"auto\")\n\ndef verify_password(plain_password: str, hashed_password: str) -> bool:\n    return pwd_context.verify(plain_password, hashed_password)\n\ndef get_password_hash(password: str) -> str:\n    return pwd_context.hash(password)\n\ndef create_access_token(\n    data: dict,\n    expires_delta: Optional[timedelta] = None\n) -> str:\n    to_encode = data.copy()\n    \n    if expires_delta:\n        expire = datetime.utcnow() + expires_delta\n    else:\n        expire = datetime.utcnow() + timedelta(\n            minutes=settings.ACCESS_TOKEN_EXPIRE_MINUTES\n        )\n    \n    to_encode.update({\"exp\": expire})\n    encoded_jwt = jwt.encode(\n        to_encode,\n        settings.SECRET_KEY,\n        algorithm=settings.ALGORITHM\n    )\n    return encoded_jwt\n\ndef decode_token(token: str) -> dict:\n    try:\n        payload = jwt.decode(\n            token,\n            settings.SECRET_KEY,\n            algorithms=[settings.ALGORITHM]\n        )\n        return payload\n    except JWTError:\n        return None\n```\n\n### 3. Authentication Routes\n\n```python\n# app/api/routes/auth.py\nfrom fastapi import APIRouter, Depends, HTTPException, status\nfrom sqlalchemy.orm import Session\nfrom datetime import timedelta\n\nfrom app.db.database import get_db\nfrom app.db.models import User\nfrom app.api.models.auth import LoginRequest, TokenResponse\nfrom app.core.security import verify_password, create_access_token, get_password_hash\nfrom app.config import settings\n\nrouter = APIRouter()\n\n@router.post(\"/login\", response_model=TokenResponse)\nasync def login(request: LoginRequest, db: Session = Depends(get_db)):\n    \"\"\"Authenticate user and return JWT token\"\"\"\n    user = db.query(User).filter(User.email == request.email).first()\n    \n    if not user or not verify_password(request.password, user.password_hash):\n        raise HTTPException(\n            status_code=status.HTTP_401_UNAUTHORIZED,\n            detail=\"Invalid credentials\"\n        )\n    \n    access_token_expires = timedelta(\n        minutes=settings.ACCESS_TOKEN_EXPIRE_MINUTES\n    )\n    access_token = create_access_token(\n        data={\"sub\": user.id},\n        expires_delta=access_token_expires\n    )\n    \n    return {\n        \"access_token\": access_token,\n        \"token_type\": \"bearer\",\n        \"expires_in\": settings.ACCESS_TOKEN_EXPIRE_MINUTES * 60\n    }\n\n@router.post(\"/register\")\nasync def register(\n    username: str,\n    email: str,\n    password: str,\n    db: Session = Depends(get_db)\n):\n    \"\"\"Register new user\"\"\"\n    #
